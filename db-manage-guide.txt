# List all employees
python manage_performance.py list

# View specific employee's performance
python manage_performance.py view EMP001

# Update employee interactively (will prompt for each metric)
python manage_performance.py update EMP001

# Set specific metric
python manage_performance.py set EMP001 --attendance 95
python manage_performance.py set EMP001 --quality 88.5

# Bulk update from CSV
python manage_performance.py bulk-update performance_data.csv

# Export all data
python manage_performance.py export monthly_report.csv

# Reset employee to defaults
python manage_performance.py reset EMP001
